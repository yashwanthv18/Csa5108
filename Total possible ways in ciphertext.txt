void encrpty(char plain[],int key){
	for (int i=1;i<26;i++)
	{
		if (plain[i]>='a'&& plain[i]<='z')
		{
			plain[i]=(plain[i]-'a'+key)%26+'a';
		}
		
	}
}

{
